void circle1(int x,int y,int xradius,int yradius)
{	
	int x1,y1;
	glColor3f(1,1,1);
	//glColor3f(0.0,0.0,0.0);
	glBegin(GL_TRIANGLE_FAN);
	for(int i=0;i<=360;i++)  //change here to affect semicircle.
	{
		float degInRad=i*DEG2RAD;
		x1=cos(degInRad)*xradius+x;
		y1=sin(degInRad)*yradius+y;
		glVertex2f(x1,y1); 
	}
	glEnd();
}

void elipsecircle(GLfloat x, GLfloat y, float xradius,float yradius)
{
	float x2, y2;
	float angle;
	glPointSize(2.0);
	glBegin(GL_TRIANGLE_FAN);
	glVertex2f(x,y);
	for (angle = 1.0; angle < 361.0f; angle += 0.2)
	{
		float degInRad = angle*DEG2RAD;
		x2 = x + sin(degInRad)*yradius;
		y2 = y + cos(degInRad)*xradius;
		glVertex2f(x2, y2);
	}
	glEnd();
}

void circle(GLfloat x, GLfloat y, GLdouble radius)
{
	float x2, y2;
	float angle;
	glPointSize(2.0);
	
	glBegin(GL_TRIANGLE_FAN);
	glVertex2f(x,y);
	for (angle = 1.0; angle < 361.0f; angle += 0.2)
	{
		x2 = x + sin(angle)*radius;
		y2 = y + cos(angle)*radius;
		glVertex2f(x2, y2);
	}
	glEnd();
}

void semicircle3(int x,int y,int xradius,int yradius)
{	
	int x1,y1;
	glColor3f(0.595,0.595,0.595);
	glBegin(GL_TRIANGLE_FAN);
	for(int i=180;i<=360;i++)  //change here to affect semicircle.
	{
		float degInRad=i*DEG2RAD;
		x1=cos(degInRad)*xradius+x;
		y1=sin(degInRad)*yradius+y;
		glVertex2f(x1,y1); 
	}
	glEnd();
}
